> nmcli radio wifi off

> sudo macchanger -r wlp3s0

> nmcli radio wifi on

> google-chrome

> code _path

> xdg-open _path

> sudo systemctl stop apache2

> sudo service mysql stop

> sudo /opt/lampp/manager-linux-x64.run

> history -c
